  // Function to fetch and update sensor readings
  function fetchSensorData() {
    var xhttp = new XMLHttpRequest();
    xhttp.onreadystatechange = function() {
      var statusElement = document.getElementById("connectionStatus");
      var statusBar = document.getElementById("esp32-status");
      
      if (this.readyState == 4) {
        if (this.status == 200) {
          try {
            var data = JSON.parse(this.responseText);
            document.getElementById("temperature").innerHTML = data.temperature.toFixed(1);
            document.getElementById("humidity").innerHTML = data.humidity.toFixed(1);
            document.getElementById("adcValue").innerHTML = data.adc;
            
            // Update Analog Chart (Progress Bar)
            var adcPercentage = (data.adc / 4095) * 100; // ESP32 ADC is 12-bit (0-4095)
            document.getElementById("adcProgressBar").style.width = adcPercentage.toFixed(0) + '%';
            document.getElementById("adcProgressBar").innerText = adcPercentage.toFixed(0) + '%';
            
            var ledStatusElement = document.getElementById("ledStatus");
            var ledIndicatorElement = document.getElementById("ledIndicator");

            if (data.led == 1) {
              ledStatusElement.innerHTML = "ON";
              ledIndicatorElement.classList.remove("off");
              ledIndicatorElement.classList.add("on");
            } else {
              ledStatusElement.innerHTML = "OFF";
              ledIndicatorElement.classList.remove("on");
              ledIndicatorElement.classList.add("off");
            }

            statusElement.innerHTML = "Online";
            statusBar.classList.remove("connecting", "error");
            statusBar.classList.add("online");

          } catch (e) {
            console.error("Error parsing JSON:", e);
            statusElement.innerHTML = "Data Error";
            statusBar.classList.remove("connecting", "online");
            statusBar.classList.add("error");
          }
        } else {
          statusElement.innerHTML = "Offline (" + this.status + ")";
          statusBar.classList.remove("connecting", "online");
            statusBar.classList.add("error");
        }
      }
    };
    xhttp.open("GET", "/readings", true);
    xhttp.send();
  }
